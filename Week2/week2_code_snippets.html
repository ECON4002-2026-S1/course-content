Note byrow = TRUE fills the matrix row by row. The default is column by column.

2.2 Matrix Multiplication

Matrix Multiplication with %*% W2-CS05

⚠ Common error Using * instead of %*% gives element-wise, not matrix, multiplication.

A <- matrix(c(1, 2, 3, 4), nrow = 2, byrow = TRUE)
B <- matrix(c(5, 6, 7, 8), nrow = 2, byrow = TRUE)

# CORRECT: matrix multiplication
A %*% B
#      [,1] [,2]
# [1,]   19   22
# [2,]   43   50
# Element [1,1] = 1×5 + 2×7 = 5 + 14 = 19

# WRONG: element-wise multiplication
A * B
#      [,1] [,2]
# [1,]    5   12
# [2,]   21   32

# Verify non-commutativity
A %*% B  # Result 1
B %*% A  # Result 2 — different!

2.3 Transpose

Transpose and the Property (AB)' = B'A' W2-CS06

Note Dimensions change from m×n to n×m. Rows become columns.

A <- matrix(c(1, 2, 3, 4, 5, 6), nrow = 2, byrow = TRUE)
t(A)
#      [,1] [,2]
# [1,]    1    4
# [2,]    2    5
# [3,]    3    6

# Verify the transpose product rule: (AB)' = B'A'
B <- matrix(c(1, 2, 3, 4, 5, 6), nrow = 3, byrow = TRUE)
all.equal(t(A %*% B), t(B) %*% t(A))  # TRUE

3. Determinants

3.1 Calculating Determinants

2×2 and 3×3 Determinants W2-CS07

Note For a 2×2 matrix: $\det\begin{bmatrix}a & b \\ c & d\end{bmatrix} = ad - bc$. A determinant of 0 means the matrix is singular (no inverse exists).

# 2×2 determinant
A <- matrix(c(3, 2, 1, 4), nrow = 2, byrow = TRUE)
det(A)   # 3×4 - 2×1 = 10

# 3×3 determinant
B <- matrix(c(1, 2, 3, 7, 0, 8, 4, 5, 6), nrow = 3, byrow = TRUE)
det(B)   # 45

# Singular matrix (det = 0)
C <- matrix(c(6, 2, 3, 1), nrow = 2, byrow = TRUE)
det(C)   # 0 — row 1 = 2 × row 2

3.2 Invertibility Check Function

is_invertible() Helper Function W2-CS08

Note A small tolerance ($10^{-10}$) handles floating-point rounding errors.

is_invertible <- function(A) {
  if (nrow(A) != ncol(A)) return(FALSE)   # Must be square
  return(abs(det(A)) > 1e-10)             # Non-zero determinant
}

A <- matrix(c(2, 1, 3, 5), nrow = 2, byrow = TRUE)
is_invertible(A)   # TRUE

B <- matrix(c(6, 2, 3, 1), nrow = 2, byrow = TRUE)
is_invertible(B)   # FALSE

4. Matrix Inverse

4.1 Finding Inverses

solve(A) and Manual 2×2 Formula W2-CS09

Note solve(A) returns $A^{-1}$. Only works if $\det(A) \neq 0$.

A <- matrix(c(2, 1, 3, 5), nrow = 2, byrow = TRUE)

# Compute inverse using R
A_inv <- solve(A)
A_inv
#            [,1]       [,2]
# [1,]  0.7142857 -0.1428571
# [2,] -0.4285714  0.2857143

# Verify: A %*% A^(-1) = I
round(A %*% A_inv, 10)

# Manual 2×2 formula: (1/det) × [[d, -b], [-c, a]]
det_A <- det(A)   # 7
A_inv_manual <- (1/det_A) * matrix(c(5, -1, -3, 2), nrow = 2, byrow = TRUE)

4.2 Safe Inversion (Error Handling)

safe_inverse() — Handles Singular Matrices W2-CS10 
safe_inverse <- function(A) {
  if (abs(det(A)) < 1e-10) {
    warning("Matrix is singular or near-singular — no inverse exists")
    return(NULL)
  }
  return(solve(A))
}

B <- matrix(c(6, 2, 3, 1), nrow = 2, byrow = TRUE)
safe_inverse(B)   # Returns NULL with a warning

5. Solving Linear Systems

5.1 Using solve()

solve(A, b) — Direct System Solver W2-CS11

Note solve(A, b) is more efficient than solve(A) %*% b — it avoids forming $A^{-1}$ explicitly.

# System: 2x + y  = 8
#          3x + 5y = 19

A <- matrix(c(2, 1, 3, 5), nrow = 2, byrow = TRUE)
b <- c(8, 19)

x <- solve(A, b)
x   # x = 3, y = 2

# Verify: A %*% x should equal b
A %*% x

5.2 Cramer's Rule

cramers_rule() Function W2-CS12

Note $x_i = \det(A_i) / \det(A)$, where $A_i$ is $A$ with column $i$ replaced by $b$.

cramers_rule <- function(A, b) {
  det_A <- det(A)
  if (abs(det_A) < 1e-10) stop("Singular matrix — no unique solution")

  n <- length(b)
  x <- numeric(n)

  for (i in 1:n) {
    A_i <- A
    A_i[, i] <- b   # Replace column i with b
    x[i] <- det(A_i) / det_A
  }
  return(x)
}

A <- matrix(c(2, 1, 3, 5), nrow = 2, byrow = TRUE)
b <- c(8, 19)
cramers_rule(A, b)   # 3, 2

5.3 Classifying Solution Types

check_system() — Unique, Infinite, or No Solution W2-CS13 
check_system <- function(A, b) {
  det_A <- det(A)

  if (abs(det_A) > 1e-10) {
    solution <- solve(A, b)
    cat("Unique solution:\n")
    print(solution)
  } else {
    augmented <- cbind(A, b)
    rank_A   <- qr(A)$rank
    rank_aug <- qr(augmented)$rank

    if (rank_A == rank_aug) {
      cat("Infinitely many solutions (dependent equations)\n")
    } else {
      cat("No solution (inconsistent system)\n")
    }
  }
}

# Test cases
A1 <- matrix(c(5, 3, 4, 1), nrow = 2, byrow = TRUE); b1 <- c(11, 6)
check_system(A1, b1)   # Unique solution

A2 <- matrix(c(2, 1, 4, 2), nrow = 2, byrow = TRUE); b2 <- c(4, 8)
check_system(A2, b2)   # Infinite solutions

A3 <- matrix(c(2, 1, 4, 2), nrow = 2, byrow = TRUE); b3 <- c(4, 10)
check_system(A3, b3)   # No solution

6. Economic Applications

6.1 Single-Market Equilibrium

solve_market_matrix() — Matrix Form of Supply & Demand W2-CS14

Economics At equilibrium $Q_d = Q_s = Q$. Rewrite both equations in the form $Q + \text{(slope)} \cdot P = \text{intercept}$, then solve as a 2×2 system.

solve_market_matrix <- function(demand_intercept, demand_slope,
                                 supply_intercept, supply_slope) {
  # Rewrite: Q + demand_slope × P = demand_intercept
  #          Q - supply_slope × P = supply_intercept
  A <- matrix(c(1,  demand_slope,
                1, -supply_slope), nrow = 2, byrow = TRUE)
  b <- c(demand_intercept, supply_intercept)

  if (abs(det(A)) < 1e-10) return(list(error = "No unique equilibrium"))

  eq <- solve(A, b)
  return(list(Q = eq[1], P = eq[2], determinant = det(A)))
}

# Qd = 100 - 2P, Qs = -20 + 4P
result <- solve_market_matrix(100, 2, -20, 4)
cat("Q* =", result$Q, "  P* =", result$P)
# Q* = 60, P* = 20

6.2 Two-Market System

Wheat and Corn Two-Market Equilibrium W2-CS15

Economics Cross-price effects appear in the off-diagonal elements of $A$. The equilibrium accounts for both markets simultaneously.

# Wheat: Qw_d = 80 - 3Pw + Pc,  Qw_s = -10 + 2Pw
# Corn:  Qc_d = 100 + Pw - 4Pc,  Qc_s = -20 + 3Pc
#
# Equilibrium (Qd = Qs):
#   Wheat:  5Pw - Pc  = 90
#   Corn:   -Pw + 7Pc = 120

A <- matrix(c(5, -1, -1, 7), nrow = 2, byrow = TRUE)
b <- c(90, 120)

prices <- solve(A, b)
Pw <- prices[1]
Pc <- prices[2]
cat("Wheat price:", round(Pw, 2))   # 21.18
cat("Corn price:",  round(Pc, 2))   # 20.18

# Equilibrium quantities
Qw <- -10 + 2 * Pw
Qc <- -20 + 3 * Pc
cat("Wheat quantity:", round(Qw, 2))  # 32.35
cat("Corn quantity:",  round(Qc, 2))  # 40.53

6.3 Leontief Input-Output Model

leontief_model() — Total Output Requirements W2-CS16

Economics The Leontief inverse $(I - A)^{-1}$ captures both direct and indirect input requirements — i.e., the full "multiplier" effect.

leontief_model <- function(A, final_demand) {
  n <- nrow(A)
  I <- diag(n)

  L <- solve(I - A)          # Leontief inverse
  x <- L %*% final_demand    # Total output

  return(list(
    leontief_inverse = L,
    total_output     = x,
    intermediate     = A %*% x,
    final            = final_demand
  ))
}

# Technology matrix (A[i,j] = input from i per unit output of j)
#                   Crops  Livestock
# Crops              0.2     0.4
# Livestock          0.1     0.2

A <- matrix(c(0.2, 0.4, 0.1, 0.2), nrow = 2, byrow = TRUE)
d <- c(80, 60)   # Final demand

result <- leontief_model(A, d)
cat("Crops total output:",     round(result$total_output[1], 1))   # 151.7
cat("Livestock total output:", round(result$total_output[2], 1))   #  96.6

7. Eigenvalues (Preview for Optimisation)

7.1 Computing Eigenvalues and Checking Definiteness

eigen() — Definiteness Check W2-CS17

Note Positive definite: all eigenvalues $> 0$ (minimum). Negative definite: all $< 0$ (maximum). Indefinite: mixed signs (saddle point).

A <- matrix(c(2, 1, 1, 3), nrow = 2, byrow = TRUE)

eig <- eigen(A)
eig$values    # 3.618, 1.382
eig$vectors   # Corresponding eigenvectors

# Check definiteness
if (all(eig$values > 0)) {
  cat("Positive definite — minimum exists\n")
} else if (all(eig$values < 0)) {
  cat("Negative definite — maximum exists\n")
} else {
  cat("Indefinite — saddle point\n")
}

7.2 Quadratic Form Analysis

quadratic_form() — Evaluate and Classify Q = x'Ax W2-CS18 
# Quadratic form: Q = 2x1² + 3x2² + 2x1x2
# Symmetric matrix form (a12 = a21 = 1)
A <- matrix(c(2, 1, 1, 3), nrow = 2, byrow = TRUE)

# Function to evaluate Q at a given x
quadratic_form <- function(A, x) {
  as.numeric(t(x) %*% A %*% x)
}

x <- c(1, 2)
quadratic_form(A, x)
# = 2(1)² + 3(2)² + 2(1)(2) = 2 + 12 + 4 = 18

# Definiteness
eig_vals <- eigen(A)$values
cat("Eigenvalues:", eig_vals)
cat("Positive definite:", all(eig_vals > 0))
